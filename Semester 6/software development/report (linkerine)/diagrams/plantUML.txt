@startuml
' Linkerine — Storage Schema (class diagram)
' Shows where each record type lives: Firestore (cloud sync) vs Dexie (local).

skinparam packageStyle rectangle

class User {
    +string uid
    +string email
}

package "Firebase Firestore (cloud, per-user, synced)" {
    class Subscription {
        +string url
        +string title
        +string folder
        +string[] tags
        +timestamp addedAt
    }

    class Bookmark {
        +string url
        +string title
        +string notes
        +timestamp dateAdded
    }

    class ReadStates {
        +string[] articles
    }

    class VaultConfig {
        +string salt
        +number iterations
        +Encrypted verifier
        +number version
        +number createdAt
    }

    class VaultItem {
        +string id
        +string iv
        +string ct
        +number createdAt
        +number updatedAt
    }
}

package "Local then Online (offline-first)" {
    class Article {
        +string id
        +string feedId
        +string title
        +string link
        +string pubDate
        +string contentSnippet
        +string content
        +number fetchedAt
    }

    class AICacheEntry {
        +string key
        +string articleId
        +string kind
        +string content
        +string provider
        +string model
        +number createdAt
    }

    class ArticleEmbedding {
        +string articleId
        +number[] vector
        +string model
        +string provider
        +number createdAt
    }
}

package "Browser-only (localStorage)" {
    class LocalSettings {
        +AISettings linkerine_ai
        +Prefs linkerine_prefs
        +CryptoKey vaultKey_inMemory
    }
}

' Relationships
User "1" --> "0..*" Subscription : Firestore
User "1" --> "0..*" Bookmark : Firestore
User "1" --> "1" ReadStates : Firestore (all + bookmarks)
User "1" --> "0..1" VaultConfig : Firestore (vault_meta)
User "1" --> "0..*" VaultItem : Firestore (vault, encrypted)

Subscription "1" --> "0..*" Article : Dexie

' Forcing AICacheEntry and ArticleEmbedding to the left of Article
Article "1" -right-> "0..1" AICacheEntry : Dexie
Article "1" -right-> "0..1" ArticleEmbedding : Dexie
AICacheEntry -[hidden]right-> LocalSettings

User ..> LocalSettings : browser only
@enduml
